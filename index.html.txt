<script>
/* hearts generator stays the same */
setInterval(() => {
  let heart = document.createElement("div");
  heart.className = "heart";
  heart.innerText = "❤";
  heart.style.left = Math.random() * 100 + "vw";
  heart.style.fontSize = (10 + Math.random()*20) + "px";
  document.body.appendChild(heart);

  setTimeout(() => heart.remove(), 6000);
}, 300);

/* questions */
let questions = [
  { q: "When did we first start talking?", a: ["May 16", "June 1st", "I forgot"], correct: 0 },
  { q: "What was our first date?", a: ["Raleys", "Movies", "Chili's"], correct: 0 },
  { q: "What do you always do?", a: ["Picking your cuticle", "Nothing", "Being quiet"], correct: 0 },
  { q: "What do I always say?", a: ["Silly", "Holy Moly", "They hit the second tower"], correct: 0 },
  { q: "What do I pretend to be mad about?", a: ["You making fun of me", "Nothing", "Other guys"], correct: 0 },
  { q: "What can we do forever?", a: ["Shop", "Sleep", "Eat"], correct: 0 },
  { q: "Who do I love the most?", a: ["Food", "Sleep", "You"], correct: 2 },
  { q: "Best thing that happened to me?", a: ["You", "School", "Life"], correct: 0 },
  { q: "Who am I choosing every time?", a: ["Nothing", "You", "Idk"], correct: 1 }
];

let current = 0;
let locked = false;

function loadQuestion() {
  let q = questions[current];
  document.getElementById("question").innerText = q.q;
  document.getElementById("progress").innerText =
    (current+1) + "/" + questions.length;

  let answersDiv = document.getElementById("answers");
  answersDiv.innerHTML = "";
  document.getElementById("result").innerText = "";
  locked = false;

  q.a.forEach((ans, index) => {
    let btn = document.createElement("button");
    btn.innerText = ans;
    btn.onclick = () => selectAnswer(index, btn);
    answersDiv.appendChild(btn);
  });
}

function selectAnswer(index, btn) {
  if (locked) return;

  let buttons = document.querySelectorAll("#answers button");

  if (index === questions[current].correct) {
    locked = true;

    // BIG correct effect
    btn.style.background = "#ff4d6d";
    btn.style.color = "black";
    btn.style.boxShadow = "0 0 20px #ff4d6d";

    document.getElementById("result").innerText = "CORRECT";

    buttons.forEach(b => b.disabled = true);

    setTimeout(() => {
      current++;
      if (current < questions.length) {
        loadQuestion();
      } else {
        endQuiz();
      }
    }, 1000);

  } else {
    // wrong = shake + try again
    btn.style.background = "#444";
    btn.style.color = "#aaa";

    document.getElementById("result").innerText = "TRY AGAIN";

    btn.animate([
      { transform: "translateX(0px)" },
      { transform: "translateX(-5px)" },
      { transform: "translateX(5px)" },
      { transform: "translateX(0px)" }
    ], { duration: 200 });

    btn.disabled = true;
  }
}

function endQuiz() {
  document.getElementById("question").innerText =
    "COMPLETE";

  document.getElementById("answers").innerHTML = "";

  let final = document.getElementById("final");

final.innerHTML = `
I’d still choose you every single time, no matter what.<br><br>

These past 10 months with you have genuinely meant so much to me, more than I probably say out loud. Every little moment we’ve had, even the small ones, are things I think about and appreciate.<br><br>

You make my life better just by being in it, and I’m really grateful for you. I like who I am when I’m with you, and I love the way we are together.<br><br>

Thank you for everything you’ve given me, for all the memories, and for just being you.<br><br>

I love you, and I’m always going to choose you.
`;

  final.style.display = "block";
}

loadQuestion();
